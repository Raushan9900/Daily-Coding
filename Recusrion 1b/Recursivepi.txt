Given a string, compute recursively a new string where all appearances of "pi" have been replaced by "3.14".
Constraints :
1 <= |S| <= 50
where |S| represents the length of string S. 
Sample Input 1 :
xpix
Sample Output :
x3.14x
Sample Input 2 :
pipi
Sample Output :
3.143.14
Sample Input 3 :
pip
Sample Output :
3.14p
Constraints:-
1<=|S|<=50


public class solution {

	// Return the changed string
	public static String replace(String input){
        String ans="";
		// Write your code here
        return replacestr(input,ans);

	}
    
    public static String replacestr(String input, String ans){
        if(input.length()<=1){
            return input;
        }
        if(input.substring(0,2).equals("pi")){
            ans=ans+"3.14"+replacestr(input.substring(2),ans);
        }
        else{
            ans=ans+input.substring(0,1)+replacestr(input.substring(1),ans);
        }
        return ans;
    }
}
